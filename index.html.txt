<!-- FIREBASE + APP -->
<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
import { getAuth, createUserWithEmailAndPassword, signInWithEmailAndPassword, signOut, onAuthStateChanged } 
from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";
import { getFirestore, doc, setDoc, getDoc, onSnapshot, updateDoc } 
from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

const firebaseConfig = {
  apiKey: "YOUR_KEY",
  authDomain: "YOUR_PROJECT.firebaseapp.com",
  projectId: "YOUR_PROJECT"
};

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);

// INPUT HELPERS
function emailInput() {
  return document.getElementById("email").value;
}

function passInput() {
  return document.getElementById("password").value;
}

// SIGNUP
window.signup = async function() {
  const userCred = await createUserWithEmailAndPassword(auth, emailInput(), passInput());

  await setDoc(doc(db, "users", userCred.user.uid), {
    tier: "Basic",
    points: 0
  });
};

// LOGIN
window.login = async function() {
  await signInWithEmailAndPassword(auth, emailInput(), passInput());
};

// LOGOUT
window.logout = async function() {
  await signOut(auth);
};

// ADD POINTS
window.addPoints = async function() {
  const uid = auth.currentUser.uid;
  const userRef = doc(db, "users", uid);

  const snap = await getDoc(userRef);
  const data = snap.data();

  const newPoints = data.points + 50;
  const newTier = getTier(newPoints);

  await updateDoc(userRef, {
    points: newPoints,
    tier: newTier
  });
};

// TIERS
function getTier(points) {
  if (points > 1000) return "Diamond";
  if (points > 700) return "Platinum";
  if (points > 500) return "Gold";
  if (points > 300) return "Silver";
  if (points > 100) return "Copper";
  return "Basic";
}

// UI UPDATE
function updateUI(user, data) {
  document.getElementById("auth").classList.add("hidden");
  document.getElementById("dashboard").classList.remove("hidden");

  document.getElementById("userEmail").innerText = user.email;
  document.getElementById("tier").innerText = data.tier + " Member";
  document.getElementById("points").innerText = data.points;

  // Tier colors
  const tierColors = {
    Basic: "#888",
    Copper: "#b87333",
    Silver: "#c0c0c0",
    Gold: "#ffd700",
    Platinum: "#e5e4e2",
    Diamond: "#00e5ff"
  };

  document.getElementById("tier").style.color = tierColors[data.tier] || "white";

  // QR Code
  QRCode.toCanvas(document.getElementById("qr"), user.uid);
}

// AUTH STATE
onAuthStateChanged(auth, (user) => {
  if (user) {
    const ref = doc(db, "users", user.uid);

    onSnapshot(ref, (docSnap) => {
      updateUI(user, docSnap.data());
    });

  } else {
    document.getElementById("auth").classList.remove("hidden");
    document.getElementById("dashboard").classList.add("hidden");
  }
});
